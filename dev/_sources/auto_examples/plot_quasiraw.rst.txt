        brainprep_quasiraw,
        brainprep_group_quasiraw,
    )
    from brainprep.config import Config
    from brainprep.reporting import RSTReport

    outdir = Path("/tmp/brainprep-quasiraw")

                output_dir=outdir,

        outputs = brainprep_group_quasiraw(
            output_dir=outdir,
